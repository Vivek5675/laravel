//in model write a:
	//accessors : accessors are change or edit the field value on UI but not change in database value.

	function getNameAttribute($val){	//funcation name should be as it change on field name and field first letter is capital complusary
		return ucfirst($val);	//this function to name first latter is capital.
	}

	function getPhoneAttribute($val){
		return "+91-".$val;
	}

	function getPhoneAttribute($val){
		$this->attributes['name']=ucfirst($val);
	}

	function getPhoneAttribute($val){
		$this->attributes['phone']="+91-".$val;
	}
